32template <typename T>
33T* align_ptr_for_type(void* destination)
34{
35 constexpr std::size_t bytes_needed{sizeof(T)};
36 constexpr std::size_t alignment{alignof(T)};
37
38 // pad the allocation for aligning the first pointer
39 auto padded_bytes_needed = bytes_needed + (alignment - 1);
40 // std::align captures last argument by reference and modifies it, but we don't want it modified
41 return reinterpret_cast<T*>(
42 std::align(alignment, bytes_needed, destination, padded_bytes_needed));
43}
