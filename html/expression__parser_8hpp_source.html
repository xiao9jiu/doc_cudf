254 void move_to_device(rmm::cuda_stream_view stream, rmm::mr::device_memory_resource* mr)
255 {
256 std::vector<cudf::size_type> sizes;
257 std::vector<void const*> data_pointers;
258
259 extract_size_and_pointer(_data_references, sizes, data_pointers);
260 extract_size_and_pointer(_literals, sizes, data_pointers);
261 extract_size_and_pointer(_operators, sizes, data_pointers);
262 extract_size_and_pointer(_operator_source_indices, sizes, data_pointers);
263
264 // Create device buffer
265 auto const buffer_size = std::accumulate(sizes.cbegin(), sizes.cend(), 0);
266 auto buffer_offsets = std::vector<int>(sizes.size());
267 thrust::exclusive_scan(sizes.cbegin(), sizes.cend(), buffer_offsets.begin(), 0);
268
269 auto h_data_buffer = std::vector<char>(buffer_size);
270 for (unsigned int i = 0; i < data_pointers.size(); ++i) {
271 std::memcpy(h_data_buffer.data() + buffer_offsets[i], data_pointers[i], sizes[i]);
272 }
273
274 _device_data_buffer = rmm::device_buffer(h_data_buffer.data(), buffer_size, stream, mr);
275
276 stream.synchronize();
277
278 // Create device pointers to components of plan
279 auto device_data_buffer_ptr = static_cast<char const*>(_device_data_buffer.data());
280 device_expression_data.data_references = device_span<detail::device_data_reference const>(
281 reinterpret_cast<detail::device_data_reference const*>(device_data_buffer_ptr +
282 buffer_offsets[0]),
283 _data_references.size());
284 device_expression_data.literals =
285 device_span<cudf::detail::fixed_width_scalar_device_view_base const>(
286 reinterpret_cast<cudf::detail::fixed_width_scalar_device_view_base const*>(
287 device_data_buffer_ptr + buffer_offsets[1]),
288 _literals.size());
289 device_expression_data.operators = device_span<ast_operator const>(
290 reinterpret_cast<ast_operator const*>(device_data_buffer_ptr + buffer_offsets[2]),
291 _operators.size());
292 device_expression_data.operator_source_indices = device_span<cudf::size_type const>(
293 reinterpret_cast<cudf::size_type const*>(device_data_buffer_ptr + buffer_offsets[3]),
294 _operator_source_indices.size());
295 device_expression_data.num_intermediates = _intermediate_counter.get_max_used();
296 shmem_per_thread = static_cast<int>(
297 (_has_nulls ? sizeof(IntermediateDataType<true>) : sizeof(IntermediateDataType<false>)) *
298 device_expression_data.num_intermediates);
299 }
