56// Creating arrow as per given type_id and buffer arguments
57template <typename... Ts>
58std::shared_ptr<arrow::Array> to_arrow_array(cudf::type_id id, Ts&&... args)
59{
60 switch (id) {
61 case type_id::BOOL8: return std::make_shared<arrow::BooleanArray>(std::forward<Ts>(args)...);
62 case type_id::INT8: return std::make_shared<arrow::Int8Array>(std::forward<Ts>(args)...);
63 case type_id::INT16: return std::make_shared<arrow::Int16Array>(std::forward<Ts>(args)...);
64 case type_id::INT32: return std::make_shared<arrow::Int32Array>(std::forward<Ts>(args)...);
65 case type_id::INT64: return std::make_shared<arrow::Int64Array>(std::forward<Ts>(args)...);
66 case type_id::UINT8: return std::make_shared<arrow::UInt8Array>(std::forward<Ts>(args)...);
67 case type_id::UINT16: return std::make_shared<arrow::UInt16Array>(std::forward<Ts>(args)...);
68 case type_id::UINT32: return std::make_shared<arrow::UInt32Array>(std::forward<Ts>(args)...);
69 case type_id::UINT64: return std::make_shared<arrow::UInt64Array>(std::forward<Ts>(args)...);
70 case type_id::FLOAT32: return std::make_shared<arrow::FloatArray>(std::forward<Ts>(args)...);
71 case type_id::FLOAT64: return std::make_shared<arrow::DoubleArray>(std::forward<Ts>(args)...);
72 case type_id::TIMESTAMP_DAYS:
73 return std::make_shared<arrow::Date32Array>(std::make_shared<arrow::Date32Type>(),
74 std::forward<Ts>(args)...);
75 case type_id::TIMESTAMP_SECONDS:
76 return std::make_shared<arrow::TimestampArray>(arrow::timestamp(arrow::TimeUnit::SECOND),
77 std::forward<Ts>(args)...);
78 case type_id::TIMESTAMP_MILLISECONDS:
79 return std::make_shared<arrow::TimestampArray>(arrow::timestamp(arrow::TimeUnit::MILLI),
80 std::forward<Ts>(args)...);
81 case type_id::TIMESTAMP_MICROSECONDS:
82 return std::make_shared<arrow::TimestampArray>(arrow::timestamp(arrow::TimeUnit::MICRO),
83 std::forward<Ts>(args)...);
84 case type_id::TIMESTAMP_NANOSECONDS:
85 return std::make_shared<arrow::TimestampArray>(arrow::timestamp(arrow::TimeUnit::NANO),
86 std::forward<Ts>(args)...);
87 case type_id::DURATION_SECONDS:
88 return std::make_shared<arrow::DurationArray>(arrow::duration(arrow::TimeUnit::SECOND),
89 std::forward<Ts>(args)...);
90 case type_id::DURATION_MILLISECONDS:
91 return std::make_shared<arrow::DurationArray>(arrow::duration(arrow::TimeUnit::MILLI),
92 std::forward<Ts>(args)...);
93 case type_id::DURATION_MICROSECONDS:
94 return std::make_shared<arrow::DurationArray>(arrow::duration(arrow::TimeUnit::MICRO),
95 std::forward<Ts>(args)...);
96 case type_id::DURATION_NANOSECONDS:
97 return std::make_shared<arrow::DurationArray>(arrow::duration(arrow::TimeUnit::NANO),
98 std::forward<Ts>(args)...);
99 default: CUDF_FAIL("Unsupported type_id conversion to arrow");
100 }
101}
