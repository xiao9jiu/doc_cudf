109constexpr multistate operator+(multistate const& lhs, multistate const& rhs)
110{
111 // combine two multistates together by full-joining LHS tails to RHS heads,
112 // and taking the corresponding LHS heads and RHS tails.
113
114 multistate result;
115 for (uint8_t lhs_idx = 0; lhs_idx < lhs.size(); lhs_idx++) {
116 auto tail = lhs.get_tail(lhs_idx);
117 for (uint8_t rhs_idx = 0; rhs_idx < rhs.size(); rhs_idx++) {
118 auto head = rhs.get_head(rhs_idx);
119 if (tail == head) { result.enqueue(lhs.get_head(lhs_idx), rhs.get_tail(rhs_idx)); }
120 }
121 }
122 return result;
123}
