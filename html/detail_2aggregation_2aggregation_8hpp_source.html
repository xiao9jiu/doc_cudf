1395template <typename F, typename... Ts>
1396CUDF_HOST_DEVICE inline decltype(auto) aggregation_dispatcher(aggregation::Kind k,
1397 F&& f,
1398 Ts&&... args)
1399{
1400 switch (k) {
1401 case aggregation::SUM:
1402 return f.template operator()<aggregation::SUM>(std::forward<Ts>(args)...);
1403 case aggregation::PRODUCT:
1404 return f.template operator()<aggregation::PRODUCT>(std::forward<Ts>(args)...);
1405 case aggregation::MIN:
1406 return f.template operator()<aggregation::MIN>(std::forward<Ts>(args)...);
1407 case aggregation::MAX:
1408 return f.template operator()<aggregation::MAX>(std::forward<Ts>(args)...);
1409 case aggregation::COUNT_VALID:
1410 return f.template operator()<aggregation::COUNT_VALID>(std::forward<Ts>(args)...);
1411 case aggregation::COUNT_ALL:
1412 return f.template operator()<aggregation::COUNT_ALL>(std::forward<Ts>(args)...);
1413 case aggregation::ANY:
1414 return f.template operator()<aggregation::ANY>(std::forward<Ts>(args)...);
1415 case aggregation::ALL:
1416 return f.template operator()<aggregation::ALL>(std::forward<Ts>(args)...);
1417 case aggregation::SUM_OF_SQUARES:
1418 return f.template operator()<aggregation::SUM_OF_SQUARES>(std::forward<Ts>(args)...);
1419 case aggregation::MEAN:
1420 return f.template operator()<aggregation::MEAN>(std::forward<Ts>(args)...);
1421 case aggregation::M2: return f.template operator()<aggregation::M2>(std::forward<Ts>(args)...);
1422 case aggregation::VARIANCE:
1423 return f.template operator()<aggregation::VARIANCE>(std::forward<Ts>(args)...);
1424 case aggregation::STD:
1425 return f.template operator()<aggregation::STD>(std::forward<Ts>(args)...);
1426 case aggregation::MEDIAN:
