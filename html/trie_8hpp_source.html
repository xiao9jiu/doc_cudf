123 struct trie_builder_node {
124 uint8_t match_length;
125 std::unordered_map<char, std::unique_ptr<trie_builder_node>> children;
126
130 void insert(std::string s) { insert(s.c_str(), s.size(), 0); }
131
132 private:
133 trie_builder_node& insert(char const* s, uint16_t size, uint8_t depth)
134 {
135 if (size == 0) {
136 match_length = depth;
137 return *this;
138 }
139
140 if (children[*s] == nullptr) { children[*s] = std::make_unique<trie_builder_node>(); }
141
142 return children[*s]->insert(s + 1, size - 1, depth + 1);
143 }
144 };

182 static trie create(std::vector<std::string> const& patterns,
183 rmm::cuda_stream_view stream,
184 rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
185 {
186 std::vector<char> tokens;
187 std::vector<uint8_t> transitions;
188 std::vector<uint8_t> match_length;
189
190 // create the trie tree
191 auto root = std::make_unique<trie_builder_node>();
192 for (auto& pattern : patterns) {
193 root->insert(pattern);
194 }
195
196 // flatten
197 auto sum = 1;
198 transitions.emplace_back(sum);
199 match_length.emplace_back(root->match_length);
200
201 auto builder_nodes = std::queue<std::unique_ptr<trie_builder_node>>();
202 builder_nodes.push(std::move(root));
203
204 tokens.emplace_back(0);
205
206 while (builder_nodes.size()) {
207 auto layer_size = builder_nodes.size();
208 for (uint32_t i = 0; i < layer_size; i++) {
209 auto node = std::move(builder_nodes.front());
210 builder_nodes.pop();
211 sum += node->children.size();
212 transitions.emplace_back(sum);
213 for (auto& item : node->children) {
214 match_length.emplace_back(item.second->match_length);
215 tokens.emplace_back(item.first);
216 builder_nodes.push(std::move(item.second));
217 }
218 }
219 }
220
221 tokens.emplace_back(0);
222
223 match_length.emplace_back(0);
224
225 std::vector<trie_node> trie_nodes;
226 auto token_counts = std::unordered_map<cudf::size_type, int32_t>();
227
228 for (uint32_t i = 0; i < tokens.size(); i++) {
229 trie_nodes.emplace_back(trie_node{tokens[i], match_length[i], transitions[i]});
230 token_counts[tokens[i]]++;
231 }
232
233 auto most_common_token =
234 std::max_element(token_counts.begin(), token_counts.end(), [](auto const& a, auto const& b) {
235 return a.second < b.second;
236 });
237
238 auto max_duplicate_tokens = most_common_token->second;
239
240 return trie{max_duplicate_tokens,
241 cudf::detail::make_device_uvector_sync(trie_nodes, stream, mr)};
242 }
243
244 [[nodiscard]] trie_device_view view() const { return trie_device_view{_nodes}; }
245};
