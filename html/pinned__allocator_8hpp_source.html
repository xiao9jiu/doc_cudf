150 __host__ inline pointer allocate(size_type cnt, const_pointer /*hint*/ = 0)
151 {
152 if (cnt > this->max_size()) { throw std::bad_alloc(); } // end if
153
154 pointer result(0);
155 CUDF_CUDA_TRY(cudaMallocHost(reinterpret_cast<void**>(&result), cnt * sizeof(value_type)));
156 return result;
157 }
