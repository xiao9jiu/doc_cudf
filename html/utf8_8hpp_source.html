62constexpr bool is_valid_begin_utf8_char(uint8_t byte)
63{
64 // to be the first byte of a valid (up to 4 byte) UTF-8 char, byte must be one of:
65 // 0b0vvvvvvv a 1 byte character
66 // 0b110vvvvv start of a 2 byte character
67 // 0b1110vvvv start of a 3 byte character
68 // 0b11110vvv start of a 4 byte character
69 return (byte & 0x80) == 0 || (byte & 0xE0) == 0xC0 || (byte & 0xF0) == 0xE0 ||
70 (byte & 0xF8) == 0xF0;
71}

79constexpr size_type bytes_in_char_utf8(char_utf8 character)
80{
81 return 1 + static_cast<size_type>((character & 0x0000'FF00u) > 0) +
82 static_cast<size_type>((character & 0x00FF'0000u) > 0) +
83 static_cast<size_type>((character & 0xFF00'0000u) > 0);
84}

96constexpr size_type bytes_in_utf8_byte(uint8_t byte)
97{
98 return 1 + static_cast<size_type>((byte & 0xF0) == 0xF0) // 4-byte character prefix
99 + static_cast<size_type>((byte & 0xE0) == 0xE0) // 3-byte character prefix
100 + static_cast<size_type>((byte & 0xC0) == 0xC0) // 2-byte character prefix
101 - static_cast<size_type>((byte & 0xC0) == 0x80); // intermediate byte
102}

111constexpr size_type to_char_utf8(const char* str, char_utf8& character)
112{
113 size_type const chr_width = bytes_in_utf8_byte(static_cast<uint8_t>(*str));
114
115 character = static_cast<char_utf8>(*str++) & 0xFF;
116 if (chr_width > 1) {
117 character = character << 8;
118 character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 8;
119 if (chr_width > 2) {
120 character = character << 8;
121 character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 16;
122 if (chr_width > 3) {
123 character = character << 8;
124 character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 24;
125 }
126 }
127 }
128 return chr_width;
129}

138constexpr inline size_type from_char_utf8(char_utf8 character, char* str)
139{
140 size_type const chr_width = bytes_in_char_utf8(character);
141 for (size_type idx = 0; idx < chr_width; ++idx) {
142 str[chr_width - idx - 1] = static_cast<char>(character) & 0xFF;
143 character = character >> 8;
144 }
145 return chr_width;
146}

155constexpr uint32_t utf8_to_codepoint(cudf::char_utf8 utf8_char)
156{
157 uint32_t unchr = 0;
158 if (utf8_char < 0x0000'0080) // single-byte pass thru
159 unchr = utf8_char;
160 else if (utf8_char < 0x0000'E000) // two bytes
161 {
162 unchr = (utf8_char & 0x1F00) >> 2; // shift and
163 unchr |= (utf8_char & 0x003F); // unmask
164 } else if (utf8_char < 0x00F0'0000) // three bytes
165 {
166 unchr = (utf8_char & 0x0F'0000) >> 4; // get upper 4 bits
167 unchr |= (utf8_char & 0x00'3F00) >> 2; // shift and
168 unchr |= (utf8_char & 0x00'003F); // unmask
169 } else if (utf8_char <= 0xF800'0000u) // four bytes
170 {
171 unchr = (utf8_char & 0x0300'0000) >> 6; // upper 3 bits
172 unchr |= (utf8_char & 0x003F'0000) >> 4; // next 6 bits
173 unchr |= (utf8_char & 0x0000'3F00) >> 2; // next 6 bits
174 unchr |= (utf8_char & 0x0000'003F); // unmask
175 }
176 return unchr;
177}

185constexpr cudf::char_utf8 codepoint_to_utf8(uint32_t unchr)
186{
187 cudf::char_utf8 utf8 = 0;
188 if (unchr < 0x0000'0080) // single byte utf8
189 utf8 = unchr;
190 else if (unchr < 0x0000'0800) // double byte utf8
191 {
192 utf8 = (unchr << 2) & 0x1F00; // shift bits for
193 utf8 |= (unchr & 0x3F); // utf8 encoding
194 utf8 |= 0x0000'C080;
195 } else if (unchr < 0x0001'0000) // triple byte utf8
196 {
197 utf8 = (unchr << 4) & 0x0F'0000; // upper 4 bits
198 utf8 |= (unchr << 2) & 0x00'3F00; // next 6 bits
199 utf8 |= (unchr & 0x3F); // last 6 bits
200 utf8 |= 0x00E0'8080;
201 } else if (unchr < 0x0011'0000) // quadruple byte utf8
202 {
203 utf8 = (unchr << 6) & 0x0700'0000; // upper 3 bits
204 utf8 |= (unchr << 4) & 0x003F'0000; // next 6 bits
205 utf8 |= (unchr << 2) & 0x0000'3F00; // next 6 bits
206 utf8 |= (unchr & 0x3F); // last 6 bits
207 utf8 |= 0xF080'8080u;
208 }
209 return utf8;
210}
