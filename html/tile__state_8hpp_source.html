28enum class scan_tile_status : uint8_t {
29 oob,
30 invalid,
31 partial,
32 inclusive,
33};

35template <typename T>
36struct scan_tile_state_view {
37 uint64_t num_tiles;
38 cuda::atomic<scan_tile_status, cuda::thread_scope_device>* tile_status;
39 T* tile_partial;
40 T* tile_inclusive;
41
42 __device__ inline void set_status(cudf::size_type tile_idx, scan_tile_status status)
43 {
44 auto const offset = (tile_idx + num_tiles) % num_tiles;
45 tile_status[offset].store(status, cuda::memory_order_relaxed);
46 }
47
48 __device__ inline void set_partial_prefix(cudf::size_type tile_idx, T value)
49 {
50 auto const offset = (tile_idx + num_tiles) % num_tiles;
51 cub::ThreadStore<cub::STORE_CG>(tile_partial + offset, value);
52 tile_status[offset].store(scan_tile_status::partial);
53 }
54
55 __device__ inline void set_inclusive_prefix(cudf::size_type tile_idx, T value)
56 {
57 auto const offset = (tile_idx + num_tiles) % num_tiles;
58 cub::ThreadStore<cub::STORE_CG>(tile_inclusive + offset, value);
59 tile_status[offset].store(scan_tile_status::inclusive);
60 }
61
62 __device__ inline T get_prefix(cudf::size_type tile_idx, scan_tile_status& status)
63 {
64 auto const offset = (tile_idx + num_tiles) % num_tiles;
65
66 while ((status = tile_status[offset].load(cuda::memory_order_relaxed)) ==
67 scan_tile_status::invalid) {}
68
69 if (status == scan_tile_status::partial) {
70 return cub::ThreadLoad<cub::LOAD_CG>(tile_partial + offset);
71 } else {
72 return cub::ThreadLoad<cub::LOAD_CG>(tile_inclusive + offset);
73 }
74 }
75};

77template <typename T>
78struct scan_tile_state {
79 rmm::device_uvector<cuda::atomic<scan_tile_status, cuda::thread_scope_device>> tile_status;
80 rmm::device_uvector<T> tile_state_partial;
81 rmm::device_uvector<T> tile_state_inclusive;
82
83 scan_tile_state(cudf::size_type num_tiles,
84 rmm::cuda_stream_view stream,
85 rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
86 : tile_status(rmm::device_uvector<cuda::atomic<scan_tile_status, cuda::thread_scope_device>>(
87 num_tiles, stream, mr)),
88 tile_state_partial(rmm::device_uvector<T>(num_tiles, stream, mr)),
89 tile_state_inclusive(rmm::device_uvector<T>(num_tiles, stream, mr))
90 {
91 }
92
93 operator scan_tile_state_view<T>()
94 {
95 return scan_tile_state_view<T>{tile_status.size(),
96 tile_status.data(),
97 tile_state_partial.data(),
98 tile_state_inclusive.data()};
99 }
100
101 inline T get_inclusive_prefix(cudf::size_type tile_idx, rmm::cuda_stream_view stream) const
102 {
103 auto const offset = (tile_idx + tile_status.size()) % tile_status.size();
104 return tile_state_inclusive.element(offset, stream);
105 }
106};

108template <typename T>
109struct scan_tile_state_callback {
110 __device__ inline scan_tile_state_callback(scan_tile_state_view<T>& tile_state,
111 cudf::size_type tile_idx)
112 : _tile_state(tile_state), _tile_idx(tile_idx)
113 {
114 }
115
116 __device__ inline T operator()(T const& block_aggregate)
117 {
118 T exclusive_prefix;
119
120 if (threadIdx.x == 0) {
121 _tile_state.set_partial_prefix(_tile_idx, block_aggregate);
122
123 auto predecessor_idx = _tile_idx - 1;
124 auto predecessor_status = scan_tile_status::invalid;
125
126 // scan partials to form prefix
127
128 auto window_partial = _tile_state.get_prefix(predecessor_idx, predecessor_status);
129 while (predecessor_status != scan_tile_status::inclusive) {
130 predecessor_idx--;
131 auto predecessor_prefix = _tile_state.get_prefix(predecessor_idx, predecessor_status);
132 window_partial = predecessor_prefix + window_partial;
133 }
134 exclusive_prefix = window_partial;
135
136 _tile_state.set_inclusive_prefix(_tile_idx, exclusive_prefix + block_aggregate);
137 }
138
139 return exclusive_prefix;
140 }
141
142 scan_tile_state_view<T>& _tile_state;
143 cudf::size_type _tile_idx;
144};
