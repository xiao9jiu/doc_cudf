48template <typename ColumnView>
49ColumnView slice(ColumnView const& input, cudf::size_type begin, cudf::size_type end)
50{
51 static_assert(std::is_same_v<ColumnView, cudf::column_view> or
52 std::is_same_v<ColumnView, cudf::mutable_column_view>,
53 "slice can be performed only on column_view and mutable_column_view");
54 CUDF_EXPECTS(begin >= 0, "Invalid beginning of range.");
55 CUDF_EXPECTS(end >= begin, "Invalid end of range.");
56 CUDF_EXPECTS(end <= input.size(), "Slice range out of bounds.");
57
58 std::vector<ColumnView> children{};
59 children.reserve(input.num_children());
60 for (size_type index = 0; index < input.num_children(); index++) {
61 children.emplace_back(input.child(index));
62 }
63
64 return ColumnView(input.type(),
65 end - begin,
66 input.head(),
67 input.null_mask(),
68 cudf::UNKNOWN_NULL_COUNT,
69 input.offset() + begin,
70 children);
71}
