46template <typename S>
47S round_up_safe(S number_to_round, S modulus)
48{
49 auto remainder = number_to_round % modulus;
50 if (remainder == 0) { return number_to_round; }
51 auto rounded_up = number_to_round - remainder + modulus;
52 if (rounded_up < number_to_round) {
53 throw std::invalid_argument("Attempt to round up beyond the type's maximum value");
54 }
55 return rounded_up;
56}
