810/*
811 * The default specialization of nullable operators is to fall back to the non-nullable
812 * implementation
813 */
814template <ast_operator op>
815struct operator_functor<op, true> {
816 using NonNullOperator = operator_functor<op, false>;
817 static constexpr auto arity = NonNullOperator::arity;
818
819 template <typename LHS,
820 typename RHS,
821 std::size_t arity_placeholder = arity,
822 std::enable_if_t<arity_placeholder == 2>* = nullptr>
823 __device__ inline auto operator()(LHS const lhs, RHS const rhs)
824 -> possibly_null_value_t<decltype(NonNullOperator{}(*lhs, *rhs)), true>
825 {
826 using Out = possibly_null_value_t<decltype(NonNullOperator{}(*lhs, *rhs)), true>;
827 return (lhs.has_value() && rhs.has_value()) ? Out{NonNullOperator{}(*lhs, *rhs)} : Out{};
828 }
829
830 template <typename Input,
831 std::size_t arity_placeholder = arity,
832 std::enable_if_t<arity_placeholder == 1>* = nullptr>
833 __device__ inline auto operator()(Input const input)
834 -> possibly_null_value_t<decltype(NonNullOperator{}(*input)), true>
835 {
836 using Out = possibly_null_value_t<decltype(NonNullOperator{}(*input)), true>;
837 return input.has_value() ? Out{NonNullOperator{}(*input)} : Out{};
838 }
839};

841// NULL_EQUAL(null, null) is true, NULL_EQUAL(null, valid) is false, and NULL_EQUAL(valid, valid) ==
842// EQUAL(valid, valid)
843template <>
844struct operator_functor<ast_operator::NULL_EQUAL, true> {
845 using NonNullOperator = operator_functor<ast_operator::NULL_EQUAL, false>;
846 static constexpr auto arity = NonNullOperator::arity;
847
848 template <typename LHS, typename RHS>
849 __device__ inline auto operator()(LHS const lhs, RHS const rhs)
850 -> possibly_null_value_t<decltype(NonNullOperator{}(*lhs, *rhs)), true>
851 {
852 // Case 1: Neither is null, so the output is given by the operation.
853 if (lhs.has_value() && rhs.has_value()) { return {NonNullOperator{}(*lhs, *rhs)}; }
854 // Case 2: Two nulls compare equal.
855 if (!lhs.has_value() && !rhs.has_value()) { return {true}; }
856 // Case 3: One value is null, while the other is not, so we return false.
857 return {false};
858 }
859};
