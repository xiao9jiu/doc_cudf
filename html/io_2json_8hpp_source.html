193 size_t get_byte_range_padding() const
194 {
195 auto const num_columns = std::visit([](const auto& dtypes) { return dtypes.size(); }, _dtypes);
196
197 auto const max_row_bytes = 16 * 1024; // 16KB
198 auto const column_bytes = 64;
199 auto const base_padding = 1024; // 1KB
200
201 if (num_columns == 0) {
202 // Use flat size if the number of columns is not known
203 return max_row_bytes;
204 }
205
206 // Expand the size based on the number of columns, if available
207 return base_padding + num_columns * column_bytes;
208 }
