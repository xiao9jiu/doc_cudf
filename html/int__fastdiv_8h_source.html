44 // Hacker's Delight, Second Edition, Chapter 10, Integer Division By Constants
45 __host__ __device__ __forceinline__ void update_magic_numbers()
46 {
47 if (d == 1) {
48 M = 0;
49 s = -1;
50 n_add_sign = 1;
51 return;
52 } else if (d == -1) {
53 M = 0;
54 s = -1;
55 n_add_sign = -1;
56 return;
57 }
58
59 int p;
60 unsigned int ad, anc, delta, q1, r1, q2, r2, t;
61 const unsigned two31 = 0x8000'0000u;
62 ad = (d == 0) ? 1 : abs(d);
63 t = two31 + ((unsigned int)d >> 31);
64 anc = t - 1 - t % ad;
65 p = 31;
66 q1 = two31 / anc;
67 r1 = two31 - q1 * anc;
68 q2 = two31 / ad;
69 r2 = two31 - q2 * ad;
70 do {
71 ++p;
72 q1 = 2 * q1;
73 r1 = 2 * r1;
74 if (r1 >= anc) {
75 ++q1;
76 r1 -= anc;
77 }
78 q2 = 2 * q2;
79 r2 = 2 * r2;
80 if (r2 >= ad) {
81 ++q2;
82 r2 -= ad;
83 }
84 delta = ad - r2;
85 } while (q1 < delta || (q1 == delta && r1 == 0));
86 this->M = q2 + 1;
87 if (d < 0) this->M = -this->M;
88 this->s = p - 32;
89
90 if ((d > 0) && (M < 0))
91 n_add_sign = 1;
92 else if ((d < 0) && (M > 0))
93 n_add_sign = -1;
94 else
95 n_add_sign = 0;
96 }

102__host__ __device__ __forceinline__ int operator/(const int n, const int_fastdiv& divisor)
103{
104 int q;
105#ifdef __CUDA_ARCH__
106 asm("mul.hi.s32 %0, %1, %2;" : "=r"(q) : "r"(divisor.M), "r"(n));
107#else
108 q = (((unsigned long long)((long long)divisor.M * (long long)n)) >> 32);
109#endif
110 q += n * divisor.n_add_sign;
111 if (divisor.s >= 0) {
112 q >>= divisor.s; // we rely on this to be implemented as arithmetic shift
113 q += (((unsigned int)q) >> 31);
114 }
115 return q;
116}
