34struct linked_column_view : public column_view_base {
35 linked_column_view(linked_column_view const&) = delete;
36 linked_column_view& operator=(linked_column_view const&) = delete;
37
38 linked_column_view(column_view const& col) : linked_column_view(nullptr, col) {}
39
40 linked_column_view(linked_column_view* parent, column_view const& col)
41 : column_view_base(col), parent(parent)
42 {
43 children.reserve(col.num_children());
44 std::transform(
45 col.child_begin(), col.child_end(), std::back_inserter(children), [&](column_view const& c) {
46 return std::make_shared<linked_column_view>(this, c);
47 });
48 }
49
50 operator column_view() const
51 {
52 auto child_it = thrust::make_transform_iterator(
53 children.begin(), [](auto const& c) { return static_cast<column_view>(*c); });
54 return column_view(this->type(),
55 this->size(),
56 this->head(),
57 this->null_mask(),
58 UNKNOWN_NULL_COUNT,
59 this->offset(),
60 std::vector<column_view>(child_it, child_it + children.size()));
61 }
62
63 linked_column_view* parent;
64 LinkedColVector children;
65};

73inline LinkedColVector table_to_linked_columns(table_view const& table)
74{
75 auto linked_it = thrust::make_transform_iterator(
76 table.begin(), [](auto const& c) { return std::make_shared<linked_column_view>(c); });
77 return LinkedColVector(linked_it, linked_it + table.num_columns());
78}
