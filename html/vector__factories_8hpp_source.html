96template <typename T>

285// Utility function template to allow copying to either a thrust::host_vector or std::vector
286template <typename T, typename OutContainer>
287OutContainer make_vector_async(device_span<T const> v, rmm::cuda_stream_view stream)
288{
289 OutContainer result(v.size());
290 CUDF_CUDA_TRY(cudaMemcpyAsync(
291 result.data(), v.data(), v.size() * sizeof(T), cudaMemcpyDeviceToHost, stream.value()));
292 return result;
293}
