126template <typename T>
127constexpr bool type_id_matches_device_storage_type(type_id id)
128{
129 return (id == type_id::DECIMAL32 && std::is_same_v<T, int32_t>) ||
130 (id == type_id::DECIMAL64 && std::is_same_v<T, int64_t>) ||
131 (id == type_id::DECIMAL128 && std::is_same_v<T, __int128_t>) || id == type_to_id<T>();
132}

437template <template <cudf::type_id> typename IdTypeMap = id_to_type_impl,
438 typename Functor,

440CUDF_HOST_DEVICE __forceinline__ constexpr decltype(auto) type_dispatcher(cudf::data_type dtype,
441 Functor f,
442 Ts&&... args)
443{
444 switch (dtype.id()) {
445 case type_id::BOOL8:
446 return f.template operator()<typename IdTypeMap<type_id::BOOL8>::type>(
447 std::forward<Ts>(args)...);
448 case type_id::INT8:
449 return f.template operator()<typename IdTypeMap<type_id::INT8>::type>(
450 std::forward<Ts>(args)...);
451 case type_id::INT16:
452 return f.template operator()<typename IdTypeMap<type_id::INT16>::type>(
453 std::forward<Ts>(args)...);
454 case type_id::INT32:
455 return f.template operator()<typename IdTypeMap<type_id::INT32>::type>(
456 std::forward<Ts>(args)...);
457 case type_id::INT64:
458 return f.template operator()<typename IdTypeMap<type_id::INT64>::type>(
459 std::forward<Ts>(args)...);
460 case type_id::UINT8:
461 return f.template operator()<typename IdTypeMap<type_id::UINT8>::type>(
462 std::forward<Ts>(args)...);
463 case type_id::UINT16:
464 return f.template operator()<typename IdTypeMap<type_id::UINT16>::type>(
465 std::forward<Ts>(args)...);
466 case type_id::UINT32:
467 return f.template operator()<typename IdTypeMap<type_id::UINT32>::type>(
468 std::forward<Ts>(args)...);
469 case type_id::UINT64:
470 return f.template operator()<typename IdTypeMap<type_id::UINT64>::type>(
471 std::forward<Ts>(args)...);
472 case type_id::FLOAT32:
473 return f.template operator()<typename IdTypeMap<type_id::FLOAT32>::type>(
474 std::forward<Ts>(args)...);
475 case type_id::FLOAT64:
476 return f.template operator()<typename IdTypeMap<type_id::FLOAT64>::type>(
477 std::forward<Ts>(args)...);
478 case type_id::STRING:
479 return f.template operator()<typename IdTypeMap<type_id::STRING>::type>(
480 std::forward<Ts>(args)...);
481 case type_id::TIMESTAMP_DAYS:
482 return f.template operator()<typename IdTypeMap<type_id::TIMESTAMP_DAYS>::type>(
483 std::forward<Ts>(args)...);
484 case type_id::TIMESTAMP_SECONDS:
485 return f.template operator()<typename IdTypeMap<type_id::TIMESTAMP_SECONDS>::type>(
486 std::forward<Ts>(args)...);
487 case type_id::TIMESTAMP_MILLISECONDS:
488 return f.template operator()<typename IdTypeMap<type_id::TIMESTAMP_MILLISECONDS>::type>(
489 std::forward<Ts>(args)...);
490 case type_id::TIMESTAMP_MICROSECONDS:
491 return f.template operator()<typename IdTypeMap<type_id::TIMESTAMP_MICROSECONDS>::type>(
492 std::forward<Ts>(args)...);
493 case type_id::TIMESTAMP_NANOSECONDS:
494 return f.template operator()<typename IdTypeMap<type_id::TIMESTAMP_NANOSECONDS>::type>(
495 std::forward<Ts>(args)...);
496 case type_id::DURATION_DAYS:
497 return f.template operator()<typename IdTypeMap<type_id::DURATION_DAYS>::type>(
498 std::forward<Ts>(args)...);
499 case type_id::DURATION_SECONDS:
500 return f.template operator()<typename IdTypeMap<type_id::DURATION_SECONDS>::type>(

502 case type_id::DURATION_MILLISECONDS:
503 return f.template operator()<typename IdTypeMap<type_id::DURATION_MILLISECONDS>::type>(
504 std::forward<Ts>(args)...);
505 case type_id::DURATION_MICROSECONDS:
506 return f.template operator()<typename IdTypeMap<type_id::DURATION_MICROSECONDS>::type>(
507 std::forward<Ts>(args)...);
508 case type_id::DURATION_NANOSECONDS:
509 return f.template operator()<typename IdTypeMap<type_id::DURATION_NANOSECONDS>::type>(
510 std::forward<Ts>(args)...);
511 case type_id::DICTIONARY32:
512 return f.template operator()<typename IdTypeMap<type_id::DICTIONARY32>::type>(
513 std::forward<Ts>(args)...);
514 case type_id::LIST:
515 return f.template operator()<typename IdTypeMap<type_id::LIST>::type>(
516 std::forward<Ts>(args)...);
517 case type_id::DECIMAL32:
518 return f.template operator()<typename IdTypeMap<type_id::DECIMAL32>::type>(

520 case type_id::DECIMAL64:
521 return f.template operator()<typename IdTypeMap<type_id::DECIMAL64>::type>(

523 case type_id::DECIMAL128:
524 return f.template operator()<typename IdTypeMap<type_id::DECIMAL128>::type>(
525 std::forward<Ts>(args)...);
526 case type_id::STRUCT:
527 return f.template operator()<typename IdTypeMap<type_id::STRUCT>::type>(
528 std::forward<Ts>(args)...);
529 default: {
